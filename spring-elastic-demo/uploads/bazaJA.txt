CREATE TABLE PROIZVOD (
ID BIGINT,
POL VARCHAR(50),
KATEGORIJA VARCHAR(50),
VRSTA_PROIZVODA VARCHAR(50),
BOJA VARCHAR(50),
CENA BIGINT,
VELICINA VARCHAR(50),
BREND VARCHAR(50),
OPIS VARCHAR(50),
NA_STANJU BOOLEAN,
TAGOVI VARCHAR(50),
SLIKA VARCHAR(50)
CONSTRAINT PROIZVOD_PK PRIMARY KEY(ID),
CONSTRAINT PROIZVOD_POL_FK FOREIGN KEY (ID) REFERENCES POL (ID_POL),
CONSTRAINT PROIZVOD_KATEGORIJA_FK FOREIGN KEY (ID) REFERENCES KATEGORIJA (ID_KATEGORIJA),
CONSTRAINT PROIZVOD_VRSTA_PROIZVODA_FK FOREIGN KEY (ID) REFERENCES VRSTA_PROIZVODA (ID_VRSTA_PROIZVODA),
CONSTRAINT PROIZVOD_BOJA_FK FOREIGN KEY (ID) REFERENCES BOJA (ID_BOJA),
CONSTRAINT PROIZVOD_VELICINA_FK FOREIGN KEY (ID) REFERENCES VELICINA (ID_VELICINA),
CONSTRAINT PROIZVOD_BREND_FK FOREIGN KEY (ID) REFERENCES BREND (ID_BREND)
);

CREATE TABLE POL (
ID_POL VARCHAR(50),
CONSTRAINT POL_PK PRIMARY KEY(ID_POL),
);

CREATE TABLE KATEGORIJA (
ID_KATEGORIJA VARCHAR(50),
CONSTRAINT KATEGORIJA_PK PRIMARY KEY(ID_KATEGORIJA),
);

CREATE TABLE VRSTA_PROIZVODA (
ID_VRSTA_PROIZVODA VARCHAR(50),
CONSTRAINT VRSTA_PROIZVODA_PK PRIMARY KEY(ID_VRSTA_PROIZVODA),
);


CREATE TABLE BOJA (
ID_BOJA VARCHAR(50),
CONSTRAINT BOJA_PK PRIMARY KEY(ID_BOJA),
);

CREATE TABLE VELICINA (
ID_VELICINA VARCHAR(50),
CONSTRAINT VELICINA_PK PRIMARY KEY(ID_VELICINA),
);

CREATE TABLE BREND (
ID_BREND VARCHAR(50),
CONSTRAINT BREND_PK PRIMARY KEY(ID_BREND),
);
